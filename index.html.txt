    <script>
        // DOM Elements
        const countdownBtn = document.getElementById('countdown-btn');
        const stopwatchBtn = document.getElementById('stopwatch-btn');
        const hoursInput = document.getElementById('hours-input');
        const minutesInput = document.getElementById('minutes-input');
        const secondsInput = document.getElementById('seconds-input');
        const startBtn = document.getElementById('start-btn');
        const pauseBtn = document.getElementById('pause-btn');
        const resetBtn = document.getElementById('reset-btn');
        const setTimeBtn = document.getElementById('set-time-btn');
        const fullscreenBtn = document.getElementById('fullscreen-btn');
        const fsStartBtn = document.getElementById('fs-start-btn');
        const fsPauseBtn = document.getElementById('fs-pause-btn');
        const fsResetBtn = document.getElementById('fs-reset-btn');
        const colorBtns = document.querySelectorAll('.color-btn');
        const presetBtns = document.querySelectorAll('.preset-btn');
        const presetTimesContainer = document.getElementById('preset-times');
        const notification = document.getElementById('notification');
        const limitNotification = document.getElementById('limit-notification');
        
        // Timer variables
        const MAX_TIME_LIMIT = 5 * 60 * 60; // 5 hours in seconds (5 * 60 * 60 = 18000)
        let timerInterval;
        let isRunning = false;
        let isCountdownMode = true;
        let timeLeft = 0; // in seconds
        let totalTime = 0; // for countdown mode
        let soundPlayed = false; // to track if sound has been played for last 5 minutes
        let isFullscreen = false;
        
        // Initialize timer
        updateInputsFromTime();
        updatePresetButtons();
        
        // Event Listeners
        countdownBtn.addEventListener('click', () => {
            isCountdownMode = true;
            countdownBtn.classList.add('active');
            stopwatchBtn.classList.remove('active');
            // Show plus time buttons in countdown mode
            presetTimesContainer.classList.remove('hidden');
            resetTimer();
        });
        
        stopwatchBtn.addEventListener('click', () => {
            isCountdownMode = false;
            stopwatchBtn.classList.add('active');
            countdownBtn.classList.remove('active');
            // Hide plus time buttons in stopwatch mode
            presetTimesContainer.classList.add('hidden');
            resetTimer();
        });
        
        startBtn.addEventListener('click', startTimer);
        pauseBtn.addEventListener('click', pauseTimer);
        resetBtn.addEventListener('click', resetTimer);
        setTimeBtn.addEventListener('click', setTimeFromInputs);
        
        // Fullscreen control buttons
        fsStartBtn.addEventListener('click', startTimer);
        fsPauseBtn.addEventListener('click', pauseTimer);
        fsResetBtn.addEventListener('click', resetTimer);
        
        fullscreenBtn.addEventListener('click', toggleFullscreen);
        
        colorBtns.forEach(btn => {
            btn.addEventListener('click', () => {
                const color = btn.getAttribute('data-color');
                document.body.style.background = `linear-gradient(135deg, ${color}, #333, #666)`;
            });
        });
        
        presetBtns.forEach(btn => {
            btn.addEventListener('click', () => {
                if (isCountdownMode && !btn.classList.contains('disabled')) {
                    const time = parseInt(btn.getAttribute('data-time'));
                    addCountdownTime(time);
                }
            });
        });

        // Input validation
        [hoursInput, minutesInput, secondsInput].forEach(input => {
            input.addEventListener('input', function() {
                // Remove non-numeric characters
                this.value = this.value.replace(/[^0-9]/g, '');
                
                // Limit values to valid ranges
                if (this.value.length > 2) {
                    this.value = this.value.slice(0, 2);
                }
                
                if (this === hoursInput && parseInt(this.value) > 99) {
                    this.value = '99';
                }
                
                if ((this === minutesInput || this === secondsInput) && parseInt(this.value) > 59) {
                    this.value = '59';
                }
            });
        });

        // Auto-advance to next input
        [hoursInput, minutesInput].forEach(input => {
            input.addEventListener('input', function() {
                if (this.value.length === 2) {
                    if (this === hoursInput) {
                        minutesInput.focus();
                    } else if (this === minutesInput) {
                        secondsInput.focus();
                    }
                }
            });
        });
        
        // Timer functions
        function startTimer() {
            if (!isRunning) {
                // If in countdown mode and time is not set, set time from inputs
                if (isCountdownMode && timeLeft === 0) {
                    setTimeFromInputs();
                    if (timeLeft === 0) return; // Don't start if still 0
                }
                
                isRunning = true;
                
                timerInterval = setInterval(() => {
                    if (isCountdownMode) {
                        if (timeLeft > 0) {
                            timeLeft--;
                            
                            // Check if 5 minutes remaining
                            if (timeLeft <= 300 && timeLeft > 0 && !soundPlayed) {
                                showNotification();
                                playSound();
                                soundPlayed = true;
                            }
                            
                            // Check if timer finished
                            if (timeLeft === 0) {
                                pauseTimer();
                                playTripleBeefSound();
                                soundPlayed = false;
                            }
                        }
                    } else {
                        // Stopwatch mode
                        timeLeft++;
                    }
                    
                    updateInputsFromTime();
                }, 1000);
            }
        }
        
        function pauseTimer() {
            clearInterval(timerInterval);
            isRunning = false;
        }
        
        function resetTimer() {
            pauseTimer();
            soundPlayed = false;
            if (isCountdownMode) {
                timeLeft = totalTime;
            } else {
                timeLeft = 0;
            }
            updateInputsFromTime();
            updatePresetButtons();
        }
        
        function setTimeFromInputs() {
            if (!isRunning && isCountdownMode) {
                const hours = parseInt(hoursInput.value) || 0;
                const minutes = parseInt(minutesInput.value) || 0;
                const seconds = parseInt(secondsInput.value) || 0;
                
                const newTime = hours * 3600 + minutes * 60 + seconds;
                
                if (newTime <= MAX_TIME_LIMIT) {
                    timeLeft = newTime;
                    totalTime = newTime;
                    soundPlayed = false;
                    removeLimitWarning();
                } else {
                    showLimitWarning();
                    // Set to maximum limit
                    timeLeft = MAX_TIME_LIMIT;
                    totalTime = MAX_TIME_LIMIT;
                    updateInputsFromTime();
                }
                updatePresetButtons();
            }
        }
        
        function addCountdownTime(seconds) {
            if (isCountdownMode) {
                const newTime = timeLeft + seconds;
                
                if (newTime <= MAX_TIME_LIMIT) {
                    timeLeft = newTime;
                    totalTime += seconds;
                    removeLimitWarning();
                } else {
                    // Set to maximum limit and show warning
                    timeLeft = MAX_TIME_LIMIT;
                    totalTime = MAX_TIME_LIMIT;
                    showLimitWarning();
                }
                updateInputsFromTime();
                updatePresetButtons();
            }
        }
        
        function updateInputsFromTime() {
            const hours = Math.floor(timeLeft / 3600);
            const minutes = Math.floor((timeLeft % 3600) / 60);
            const seconds = timeLeft % 60;
            
            hoursInput.value = hours.toString().padStart(2, '0');
            minutesInput.value = minutes.toString().padStart(2, '0');
            secondsInput.value = seconds.toString().padStart(2, '0');
            
            // Change color when less than 5 minutes in countdown mode
            if (isCountdownMode && timeLeft <= 300 && timeLeft > 0) {
                hoursInput.style.color = '#ff5252';
                minutesInput.style.color = '#ff5252';
                secondsInput.style.color = '#ff5252';
            } else {
                hoursInput.style.color = 'white';
                minutesInput.style.color = 'white';
                secondsInput.style.color = 'white';
            }
            
            // Show warning style when at maximum limit
            if (timeLeft >= MAX_TIME_LIMIT) {
                hoursInput.classList.add('max-limit');
                minutesInput.classList.add('max-limit');
                secondsInput.classList.add('max-limit');
            } else {
                hoursInput.classList.remove('max-limit');
                minutesInput.classList.remove('max-limit');
                secondsInput.classList.remove('max-limit');
            }
        }
        
        function updatePresetButtons() {
            presetBtns.forEach(btn => {
                const timeToAdd = parseInt(btn.getAttribute('data-time'));
                if (timeLeft + timeToAdd > MAX_TIME_LIMIT) {
                    btn.classList.add('disabled');
                } else {
                    btn.classList.remove('disabled');
                }
            });
        }

        function playSound() {
            // Create a simple beep sound for 5-minute warning
            const audioContext = new (window.AudioContext || window.webkitAudioContext)();
            const oscillator = audioContext.createOscillator();
            const gainNode = audioContext.createGain();
            
            oscillator.connect(gainNode);
            gainNode.connect(audioContext.destination);
            
            oscillator.frequency.value = 600;
            oscillator.type = 'sine';
            
            gainNode.gain.setValueAtTime(0.3, audioContext.currentTime);
            gainNode.gain.exponentialRampToValueAtTime(0.01, audioContext.currentTime + 0.8);
            
            oscillator.start(audioContext.currentTime);
            oscillator.stop(audioContext.currentTime + 0.8);
        }

        function toggleFullscreen() {
            if (!document.fullscreenElement) {
                document.documentElement.requestFullscreen().catch(err => {
                    console.error(`Error attempting to enable fullscreen: ${err.message}`);
                });
                document.body.classList.add('fullscreen-mode');
                fullscreenBtn.textContent = 'Exit Fullscreen';
                isFullscreen = true;
            } else {
                if (document.exitFullscreen) {
                    document.exitFullscreen();
                    document.body.classList.remove('fullscreen-mode');
                    fullscreenBtn.textContent = 'Fullscreen';
                    isFullscreen = false;
                }
            }
        }
        
        // Handle fullscreen change events
        document.addEventListener('fullscreenchange', () => {
            if (!document.fullscreenElement) {
                document.body.classList.remove('fullscreen-mode');
                fullscreenBtn.textContent = 'Fullscreen';
                isFullscreen = false;
            }
        });
